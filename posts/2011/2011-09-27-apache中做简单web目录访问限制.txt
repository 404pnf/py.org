# apache中做简单web目录访问限制

[tags]apache[/tags]

	# Authorize for setup
	<Directory /usr/share/phpmyadmin/setup>
	    <IfModule mod_authn_file.c>
	    AuthType Basic
	    AuthName "phpMyAdmin Setup"
	    AuthUserFile /etc/phpmyadmin/htpasswd.setup
	    </IfModule>
	    Require valid-user
	</Directory>
